import torch

# 返回当前设备索引
torch.cuda.current_device()

# 返回gpu数量
torch.cuda.device_count()

# 返回gpu名称，索引从0开始
torch.cuda.get_device_name(0)

# cuda是否可用
torch.cuda.is_available()
